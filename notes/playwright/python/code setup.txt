#! sync methode - run severything in order

    from playwright.sync_api import sync_playwright = imports sync_playwright from the library

    with sync_playwright() as p: = creates an alias 'p' for the import
        browser = p.chromium.launch(headless=False) = sets the browser to Chromium
        page = browser.new_page() = opens a new page
        page.goto("https://www.saucedemo.com/") = navigates to this page

#! async methode - runs everything independent from each other, unless you add await which creates an order

    import asyncio
    from playwright.async_api import async_playwright

    async def main():
        async with async_playwright() as p:
            browser = await p.firefox.launch(headless = False)
            page = await browser.new_page()
            await page.goto("https://www.saucedemo.com/")
            print (await page.title())

    asyncio.run(main())
